Основной сервис интеграции с 1С

//запишем сервис в переменную, чтобы с ним легче было работать
var service = Locator.GetServiceNotNull\<Integration1CService\>();
//Чтобы получить соединения с 1С, используйте следующие методы:
//1. GetConnections() - получить коллекцию соединений
var connectionCollection = service.GetConnections();
//2. GetConnection(configName) - получить соединение по имени
var connection = service.GetConnection(configName);
//Чтобы проверить соединение, используйте следующие функции:
//1. CheckConnection(connection) - передается проверяемое соединение
service.CheckConnection(connection);
//2. CheckConnectionConfiguration(configName) - передается имя соединения
var check = service.CheckConnectionConfiguration(configName);
//Чтобы получить полную конфигурацию, используйте функцию GetFullConfiguration(configName)
var fullConfiguration = service.GetFullConfiguration(configName);
//Чтобы получить текущую конфигурацию, используйте функцию GetCurrentConfiguration(configName)
var curConfig = service.GetCurrentConfiguration(configName);
//Для управления конфигурацией используйте следующие функции:
//1. EnableConfiguration(configName) - включить конфигурацию
service.EnableConfiguration(configName);
//2. DisableConfiguration(configName) - выключить конфигурацию
service.DisableConfiguration(configName);
//3. RebootConfiguration(configName) - перезагрузить конфигурацию
service.RebootConfiguration(configName);
//Чтобы получить коннектор, используйте функцию GetComConnector(configName)
var connector = service.GetComConnector(configName);
//Чтобы получить провайдер, используйте функцию GetProvider(configName)
var provider = service.GetProvider(configName);
//Чтобы получить импортированные из 1C объекты, используйте функцию LoadConfObject<T>(uid)
//В данном случае загружаем контрагента
var contractor1C = service.LoadConfObject\<EleWise.ELMA.Integration1C.Configs.One.Kontragenty\>(uid);
