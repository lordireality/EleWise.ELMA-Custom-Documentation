Detailed Description

Сервис выполнения команды процессу массового импорта данных

//При помощи данного сервиса можно управлять запущенным процессом массового импорта
//Чтобы отправить команду процессу импорта используйте ExecuteCommand(workUid, command), где
//<c>Guid</c> workUid - уникальный идентификатор процесса массового импорта;
//<c>ImportCommand</c> command - выполняемая команда;
PublicAPI.Services.BulkDataImportCommand.SetCommand(workUid, command);
//Чтобы получить информацию по запущенному импорту GetCommand(workUid), где
//<c>Guid</c> workUid - уникальный идентификатор процесса массового импорта;
//Возвращает сохранённую команду <c>ImportCommand?</c> - null, если нет сохранённой команды.
PublicAPI.Services.BulkDataImport.GetCommand(workUid);
//Чтоб удалить текущую сохранённую команду, используйте метод Reset(workUid), где
//<c>Guid</c> workUid - уникальный идентификатор процесса массового импорта;
PublicAPI.Services.BulkDataImport.Reset(workUid);
