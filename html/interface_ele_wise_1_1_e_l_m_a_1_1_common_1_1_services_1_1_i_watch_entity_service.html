Detailed Description

Управление списками наблюдений

//С помощью данного сервиса мы можем управлять списками наблюдателей:
//получать списки наблюдателей, очищать списки наблюдателей и создавать наблюдателей
//Чтобы получить список наблюдателей, используйте функцию GetWatches(objectUid, actionUid, entity), где:
//Guid objectUid - идентификатор типа сущности;
//Guid actionUid - идентификатор действия;
//IEntity entity - сущность
var userList = Locator.GetServiceNotNull\<IWatchEntityService\>().GetWatches(objectUid, actionUid, entity);
//Чтобы очистить список наблюдателей для определённого объекта используйте функцию ClearForObject(objectUid, entityId).
//Чтобы очистить список наблюдателей для типа объекта вместо entityId передавайте значение null.
Locator.GetServiceNotNull\<IWatchEntityService\>().ClearForObject(objectUid, entityId)
//Чтобы добавить пользователя в список наблюдателей используйте функцию CreateWatchByUser(typeUid, entityId, user), где
//Guid typeUid - идентификатор типа сущности;
//long? entityId - идентификатор сущности;
//IUser user - пользователь.
Locator.GetServiceNotNull\<IWatchEntityService\>().CreateWatchByUser(typeUid, entityId, user);
