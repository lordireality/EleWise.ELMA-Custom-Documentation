Detailed Description

Точка расширения для вычисления уровня совпадения текущего адреса страницы с элементом меню &.

При указании в параметре CurrentMatcher атрибута ContentItemAttribute типа на реализацию данной точки расширения, будет использовать метод MatchCurrent для вычисления уровня совпадения текущего адреса страницы с элементом меню

[Component] class IncomingMatcher : IContentItemCurrentMatcher { public int MatchCurrent(RequestContext requestContext) { if (Equals(requestContext.RouteData.Values["action"], "Incoming") && Equals(requestContext.RouteData.Values["controller"], "AllTasks") && Equals(requestContext.RouteData.Values["area"], RouteProvider.AreaName)) return (int)ContentItemMatchLevel.AreaControllerAction; if (Equals(requestContext.RouteData.Values["controller"], "AllTasks") && Equals(requestContext.RouteData.Values["area"], RouteProvider.AreaName)) return (int)ContentItemMatchLevel.AreaController; if (Equals(requestContext.RouteData.Values["controller"], "Task") && Equals(requestContext.RouteData.Values["area"], RouteProvider.AreaName)) return (int)ContentItemMatchLevel.AreaPlus; if (Equals(requestContext.RouteData.Values["area"], RouteProvider.AreaName)) return (int)ContentItemMatchLevel.Area; return (int)ContentItemMatchLevel.Nothing; } }
