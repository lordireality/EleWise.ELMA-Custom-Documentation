Detailed Description

Сервис для работы с сообщениями в процессах Workflow.

//С помощью данного сервиса можно получить информацию об активынх точках останова экземпляров процессов
//на операции "Ожидание сообщения" и отправлять сообщения в процессы
//запишем сервис в переменную, чтобы с ним легче было работать
var service = Locator.GetServiceNotNull\<IWorkflowMessagingService\>();
//Чтобы получить информацию об активных точках останова экземпляров процессов на операции "Ожидание сообщения" используйте следующие функции:
//1.GetActiveMessageBookmarkInfos(instanceList) - проверяет только по процессам, записанным в переменную
var info = services.GetActiveMessageBookmarkInfos(instanceList);
//2.GetActiveMessageBookmarkInfos(instanceList, allowSubInstances) - проверяет с учётом подпроцессов
var info = services.GetActiveMessageBookmarkInfos(instanceList, allowSubInstances);
//Чтобы отправить сообщение, используйте следующие функции:
//1.SendMessage(messageType, data) - отправить сообщение
var sendResult = service.SendMessage(messageType, data);
//2.SendMessage(instance, messageType, data) - отправить сообщение в указанный экземпляр
var sendResult = service.SendMessage(instance, messageType, data);
